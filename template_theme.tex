\definecolorstyle{myColorStyle}{}
{
  % Background Colors
  \colorlet{backgroundcolor}{\fontcolor}
  \colorlet{framecolor}{\fontcolor}
  % Title Colors
  \colorlet{titlefgcolor}{\fontcolor}
  \colorlet{titlebgcolor}{\secondcolor!80}
  % Block Colors
  \colorlet{blocktitlebgcolor}{\maincolor}
  \colorlet{blocktitlefgcolor}{\fontcolor}
  \colorlet{blockbodybgcolor}{\fontcolor}
  \colorlet{blockbodyfgcolor}{black}
  % Innerblock Colors
  \colorlet{innerblocktitlebgcolor}{\maincolor}
  \colorlet{innerblocktitlefgcolor}{black}
  \colorlet{innerblockbodybgcolor}{\fontcolor}
  \colorlet{innerblockbodyfgcolor}{black}
  % Note colors
  \colorlet{notefgcolor}{black}
  \colorlet{notebgcolor}{\secondcolor!10}
  \colorlet{notefrcolor}{\fontcolor}
}
\usecolorstyle{myColorStyle}


\definebackgroundstyle{myBackgroundStyle}{}
{
}
\usebackgroundstyle{myBackgroundStyle}


\defineblockstyle{myBlockStyle}{roundedcorners=20}
{
  % Block body style
  \draw[color=blockbodybgcolor,
        fill=blockbodybgcolor,
        rounded corners=\blockroundedcorners]
  (blockbody.south west) rectangle (blockbody.north east);
  % Block title style
  \ifBlockHasTitle
    %\draw[color=blocktitlebgcolor,
    %      fill=blocktitlebgcolor,
    %      rounded corners=\blockroundedcorners]
    \shade[left color=blocktitlebgcolor,
           right color=blocktitlebgcolor!80,
           rounded corners=\blockroundedcorners]
    (blocktitle.south west) rectangle (blocktitle.north east);
  \fi
}
\useblockstyle{myBlockStyle}


\defineinnerblockstyle{myInnerBlockStyle}{}
{
  % Inner block body style
  \draw[color=innerblockbodybgcolor,
        fill=innerblockbodybgcolor,
        rounded corners=\innerblockroundedcorners]
        (innerblockbody.south west) rectangle (innerblockbody.north east);
  % Inner block title style
  \ifBlockHasTitle
    \draw[color=innerblocktitlebgcolor,
          fill=innerblocktitlebgcolor,
          dashed,
          line width=3pt]
    (innerblocktitle.south west) -- (innerblocktitle.south east);
  \fi
}
\useinnerblockstyle{myInnerBlockStyle}


% Title style
\newcommand{\titlewidthrate}{.7}
\definetitlestyle{myTitleStyle}
{
  width=\linewidth,
  roundedcorners=20,
  linewidth=2pt,
  innersep=20pt,
  titletotopverticalspace=15mm,
  titletoblockverticalspace=30mm
}{
  \begin{scope}[line width=\titlelinewidth,
                rounded corners=\titleroundedcorners]
    %\draw[color=titlebgcolor,
    %      fill=titlebgcolor]
    \shade[left color=titlebgcolor, right color=titlebgcolor!70]
    (\titlewidthrate\titleposleft,\titleposbottom) rectangle (\titlewidthrate\titleposright,\titlepostop);
  \end{scope}
}
\usetitlestyle{myTitleStyle}
